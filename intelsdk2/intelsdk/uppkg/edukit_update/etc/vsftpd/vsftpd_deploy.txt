1.将配置文件目录放置路径  
	/etc/vsftpd.conf

2.部署FTP服务器
	mv /etc/vsftpd/vsftpd /usr/local/sbin/vsftpd
	mv /etc/vsftpd/vsftpd.conf /etc/vsftpd.conf
3.创建用户
	addgroup ftpgroup	
	adduser  admin
	passwd admin
	adduser  guest
	passwd guest
4.创建目录
	mkdir /usr/share/empty
	mkdir -p /opt/Rec
	chmod 755 /opt/Rec
	

5. 运行可在任意目录下执行
	vsftpd
